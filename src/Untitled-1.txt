 "
        #version 330 core
        in vec2 position;
        in vec2 texture_coords;
        
        uniform mat4 matrix;
        out vec2 v_tex_coords;

        void main() {
            gl_Position = matrix * vec4(position, 0.0, 1.0);
            
            v_tex_coords = texture_coords;
        }
        ",
        "
        #version 330 core
        in vec2 v_tex_coords;
        out vec4 FragColor;

        uniform sampler2D tex;

        void main() {
            FragColor = texture(tex, v_tex_coords);
        }
        ",
